Create EarthquakeViz.jsx:

```jsx
import { useEffect, useRef, useState } from 'react'
import * as d3 from 'd3'
import earthquakeData from '../data/earthquakes.json'
import './EarthquakeViz.css'

function EarthquakeViz() {
  const svgRef = useRef(null)
  const [tooltip, setTooltip] = useState({ show: false, x: 0, y: 0, content: '' })
  const [dimensions, setDimensions] = useState({ width: 600, height: 400 })

  useEffect(() => {
    // Responsive resize
    const handleResize = () => {
      const container = svgRef.current?.parentElement
      if (container) {
        setDimensions({
          width: container.clientWidth,
          height: Math.min(400, container.clientWidth * 0.6)
        })
      }
    }
    
    handleResize()
    window.addEventListener('resize', handleResize)
    return () => window.removeEventListener('resize', handleResize)
  }, [])

  useEffect(() => {
    if (!svgRef.current) return

    const svg = d3.select(svgRef.current)
    svg.selectAll('*').remove()

    const margin = { top: 20, right: 30, bottom: 50, left: 50 }
    const width = dimensions.width - margin.left - margin.right
    const height = dimensions.height - margin.top - margin.bottom

    const g = svg
      .attr('width', dimensions.width)
      .attr('height', dimensions.height)
      .append('g')
      .attr('transform', `translate(${margin.left},${margin.top})`)

    // Scales
    const xScale = d3.scaleLinear()
      .domain([0, d3.max(earthquakeData, d => d.depth) + 10])
      .range([0, width])

    const yScale = d3.scaleLinear()
      .domain([0, d3.max(earthquakeData, d => d.magnitude) + 1])
      .range([height, 0])

    const colorScale = d3.scaleSequential(d3.interpolateReds)
      .domain([3, 7])

    // Axes
    g.append('g')
      .attr('transform', `translate(0,${height})`)
      .call(d3.axisBottom(xScale))
      .append('text')
      .attr('x', width / 2)
      .attr('y', 40)
      .attr('fill', '#333')
      .attr('text-anchor', 'middle')
      .text('Depth (km)')

    g.append('g')
      .call(d3.axisLeft(yScale))
      .append('text')
      .attr('transform', 'rotate(-90)')
      .attr('y', -40)
      .attr('x', -height / 2)
      .attr('fill', '#333')
      .attr('text-anchor', 'middle')
      .text('Magnitude')

    // Points
    g.selectAll('circle')
      .data(earthquakeData)
      .join('circle')
      .attr('cx', d => xScale(d.depth))
      .attr('cy', d => yScale(d.magnitude))
      .attr('r', 8)
      .attr('fill', d => colorScale(d.magnitude))
      .attr('opacity', 0.7)
      .style('cursor', 'pointer')
      .on('mouseenter', (event, d) => {
        const [x, y] = d3.pointer(event, svgRef.current)
        setTooltip({
          show: true,
          x: x + 10,
          y: y - 10,
          content: `${d.location}\nM${d.magnitude} | ${d.depth}km\n${d.date}`
        })
      })
      .on('mouseleave', () => {
        setTooltip({ show: false, x: 0, y: 0, content: '' })
      })

  }, [dimensions])

  return (
    <div className="earthquake-viz">
      <h3>Indonesian Earthquakes (2026)</h3>
      <p className="viz-description">
        Interactive scatterplot showing earthquake magnitude vs depth
      </p>
      <svg ref={svgRef}></svg>
      {tooltip.show && (
        <div 
          className="tooltip" 
          style={{ left: tooltip.x, top: tooltip.y }}
        >
          {tooltip.content.split('\n').map((line, i) => (
            <div key={i}>{line}</div>
          ))}
        </div>
      )}
      <div className="legend">
        <span>Lower Magnitude</span>
        <div className="gradient-bar"></div>
        <span>Higher Magnitude</span>
      </div>
    </div>
  )
}

export default EarthquakeViz
```

```jsx
import { useEffect, useRef, useState } from 'react'
import * as d3 from 'd3'
import './ResearchDataViz.css'

// Sample seismic data
const seismicData = [
  { velocity: 3.5, depth: 0 },
  { velocity: 4.2, depth: 5 },
  { velocity: 5.1, depth: 10 },
  { velocity: 5.8, depth: 15 },
  { velocity: 6.2, depth: 20 },
  { velocity: 6.5, depth: 25 },
  { velocity: 6.8, depth: 30 },
]

function ResearchDataViz() {
  const svgRef = useRef(null)
  const [dimensions, setDimensions] = useState({ width: 600, height: 300 })

  useEffect(() => {
    const handleResize = () => {
      const container = svgRef.current?.parentElement
      if (container) {
        setDimensions({
          width: container.clientWidth,
          height: Math.min(300, container.clientWidth * 0.5)
        })
      }
    }
    
    handleResize()
    window.addEventListener('resize', handleResize)
    return () => window.removeEventListener('resize', handleResize)
  }, [])

  useEffect(() => {
    if (!svgRef.current) return

    const svg = d3.select(svgRef.current)
    svg.selectAll('*').remove()

    const margin = { top: 20, right: 30, bottom: 40, left: 50 }
    const width = dimensions.width - margin.left - margin.right
    const height = dimensions.height - margin.top - margin.bottom

    const g = svg
      .attr('width', dimensions.width)
      .attr('height', dimensions.height)
      .append('g')
      .attr('transform', `translate(${margin.left},${margin.top})`)

    // Scales
    const xScale = d3.scaleLinear()
      .domain([0, d3.max(seismicData, d => d.depth)])
      .range([0, width])

    const yScale = d3.scaleLinear()
      .domain([0, d3.max(seismicData, d => d.velocity) + 1])
      .range([height, 0])

    // Axes
    g.append('g')
      .attr('transform', `translate(0,${height})`)
      .call(d3.axisBottom(xScale))
      .append('text')
      .attr('x', width / 2)
      .attr('y', 35)
      .attr('fill', '#333')
      .attr('text-anchor', 'middle')
      .text('Depth (km)')

    g.append('g')
      .call(d3.axisLeft(yScale))
      .append('text')
      .attr('transform', 'rotate(-90)')
      .attr('y', -40)
      .attr('x', -height / 2)
      .attr('fill', '#333')
      .attr('text-anchor', 'middle')
      .text('Seismic Velocity (km/s)')

    // Line
    const line = d3.line()
      .x(d => xScale(d.depth))
      .y(d => yScale(d.velocity))
      .curve(d3.curveMonotoneX)

    g.append('path')
      .datum(seismicData)
      .attr('fill', 'none')
      .attr('stroke', '#2563eb')
      .attr('stroke-width', 2)
      .attr('d', line)

    // Points
    g.selectAll('circle')
      .data(seismicData)
      .join('circle')
      .attr('cx', d => xScale(d.depth))
      .attr('cy', d => yScale(d.velocity))
      .attr('r', 4)
      .attr('fill', '#2563eb')

  }, [dimensions])

  return (
    <div className="research-viz">
      <h3>Seismic Velocity Profile</h3>
      <p className="viz-description">
        P-wave velocity increase with depth
      </p>
      <svg ref={svgRef}></svg>
    </div>
  )
}

export default ResearchDataViz
```
